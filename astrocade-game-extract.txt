    let commitTimer = null;
    let isDragging = false;
    let isBatchOperation = false;
    // Debounced commit function - delays commit until user stops making changes
    function debouncedCommit(delay = 500) {
        if (commitTimer !== null) {
            clearTimeout(commitTimer);
        }
        commitTimer = setTimeout(() =&gt; {
            if (currentMode === 'edit' &amp;&amp; !isDragging &amp;&amp; !isBatchOperation) {
                commitState();
            }
            commitTimer = null;
        }, delay);
    }
